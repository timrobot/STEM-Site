# database crawled, prefixtree index
def crawl_web(seed, words, crawled, index):
to_crawl = [seed]
i = 0
while (i < len(to_crawl):
if (!crawled.lookup(to_crawl[i])):
crawled.add(get_links(to_crawl[i]))
for word in words:
page = get_page(to_crawl[i])
index.insert(word, (to_crawl[i], get_word_frequency(page, word)))
i++
